# useSignTransaction

Hook for signing a transaction.

## Import

```ts
import { useSignTransaction } from 'wagmi'
```

## Usage

::: code-group
```tsx [index.tsx]
import { useSignTransaction } from 'wagmi'
import { parseEther } from 'viem'

function App() {
  const { signTransaction } = useSignTransaction()

  return (
    <button
      onClick={() =>
        signTransaction({
          to: '0xd2135CfB216b74109775236E36d4b433F1DF507B',
          value: parseEther('0.01'),
        })
      }
    >
      Sign transaction
    </button>
  )
}
```
