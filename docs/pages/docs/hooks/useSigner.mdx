# useSigner

Hook for accessing [`Signer`](https://docs.ethers.io/v5/api/signer) object for connected account.

```tsx
import { useSigner } from 'wagmi'
```

## Usage

```tsx
import { useContract, useSigner } from 'wagmi'

const App = () => {
  const [{ data: signer, error, loading }, getSigner] = useSigner()

  const contract = useContract({
    addressOrName: '0xa50f703771d04d9e76e03b20720539fc014aaa40',
    contractInterface: ERC20ABI,
    signerOrProvider: signer,
  })

  return ...
}
```

## Return Values

### result

```ts
{
  data?: ethers.Signer
  error?: Error
  loading?: boolean
}
```

### getSigner

```ts
() => Promise<ethers.Signer>
```
