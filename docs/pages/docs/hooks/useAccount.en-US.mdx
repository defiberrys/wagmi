# useAccount

Hook for accessing the connected account.

```ts
import { useAccount } from 'wagmi'
```

## Usage

```tsx
import { useAccount } from 'wagmi'

function App() {
  const { data, disconnect, isError, isLoading } = useAccount()

  if (isLoading) return <div>Loading accountâ€¦</div>
  if (isError) return <div>Error loading account</div>
  return (
    <>
      <div>{data?.address}</div>

      <button onClick={disconnect}>
        Disconnect from {data?.connector?.name}
      </button>
    </>
  )
}
```

## Return Values

```tsx
{
  data?: {
    address: string
    connector: Connector
    ens?: {
      avatar?: string | null
      name?: string | null
    }
  }
  error?: Error
  isError boolean
  isFetching: boolean
  isIdle: boolean
  isLoading: boolean
  isSuccess: boolean
  status: "idle" | "error" | "loading" | "success"
}
```

## Configuration

### ens (optional)

Fetches ENS information for connected account. When `true`, fetches both ENS name and avatar. Also, accepts object in the shape of `{ avatar?: boolean; name?: boolean }`.

```tsx {5}
import { useAccount } from 'wagmi'

function App() {
  const account = useAccount({
    ens: { name: true },
  })
}
```

### suspense (optional)

Set this to `true` to enable suspense mode.

```tsx {5}
import { useAccount } from 'wagmi'

function App() {
  const balance = useAccount({
    suspense: true,
  })
}
```
